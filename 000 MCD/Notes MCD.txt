Description des tables

	Table SkyObjects : listes des objets dans le ciel : etoiles, galaxies, amas, nebuleuses ...

		dbId				primary key									integer
		hipId				hipparcos catalog id 						varchar(255)
		hdId				Henry Draper Catalog id 					varchar(255)
		hrId 				harvard catalog id  						varchar(255)
		glId 				gliese id 									varchar(255)
		bfId	   			bayer/flamsteed id 							varchar(255)
		rightAsc			right ascension 							varchar(50)
		declination			declination 								varchar(50)
		properName 			common name									varchar(255)	
		typeId 				type of the object star, planet, ...									
		distancePC 			distance in parsec 							numeric(,)
		radialVelocity 		radial velocity 							numeric(,)
		visualMagnitude 	visual magnitude 							numeric(6,4)
		absoluteMagnitude 	apparent magnitude at 10 pc 				numeric(5,5)
		spectralType 		spectral type 								varchar(10)
		ColorIndex 			color index 								numeric(,)
		constellationId 	constellation abbreviation 					varchar(5)
		luminosity 			stars luminosity	 						numeric(,)
		variableStarLabel	standard variable star designation  		varchar(20)
		variableMinMag 		mag min for var stars 						numeric(6,4)
		variableMaxMag 		mag max for var stars 						numeric(6,4)
		Table category : type of stellar object
		description 		description of the object 					varchar(1024)
		primaryStarId 		id of primary star 							integer
		companionStarId		id of companion star 						integer
